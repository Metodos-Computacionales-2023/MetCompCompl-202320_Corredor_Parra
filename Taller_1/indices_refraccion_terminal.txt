
while IFS=',' read -r categoria fabricante material enlace; do
   
    categoria=$(echo "$categoria" | xargs)
    enlace=$(echo "$enlace" | xargs)

    
    if [ ! -d "$categoria" ]; then
        mkdir "$categoria"
    fi
    

    wget -P "$categoria" -O "$categoria/$material" "$enlace"
done < indices_refraccion.csv
